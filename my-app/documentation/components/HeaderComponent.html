import { Component, Input, OnInit } from '@angular/core';
import { OwlOptions } from 'ngx-owl-carousel-o';

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss']
})
export class HeaderComponent implements OnInit {

  constructor() { }

  ngOnInit(): void {
  }
  imgPrefix: string = `https://image.tmdb.org/t/p/w500`;
  @Input() headerTrendingMovie: any[] = [];

  customOptions: OwlOptions = {
    loop: true,
    mouseDrag: true,
    touchDrag: false,
    pullDrag: false,
    dots: false,
    navSpeed: 800,
    navText: ['', ''],
    responsive: {
      0: {
        items: 1
      },
      400: {
        items: 2
      },
      740: {
        items: 3
      },
      940: {
        items: 6
      }
    },
    nav: true
  }

}

    <owl-carousel-o [options]="customOptions" class="d-flex justify-content-center align-items-center  pt-5 px-2">
      <ng-container  *ngFor="let movie of headerTrendingMovie">
        <ng-template  carouselSlide [id]="movie.id">
          <img class="w-100 " [src]="imgPrefix+movie.poster_path" >

</div>
